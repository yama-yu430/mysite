    public void showStock(){
        System.out.println("スナック自販機 - 在庫" + getStock() + "個、価格：" + getPrice() + "円、商品：" + snackType);
    }
    @Override
    public void buy(){
        int stock = getStock();
        if(stock >=0){
            setStock(--stock);
            System.out.println("スナック自販機 - 商品(" + snackType + ")購入しました");
        } else {
